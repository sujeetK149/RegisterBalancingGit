4def reconstruct_expression(node,internalVariables):
5 """
6 @brief Recursively reconstructs expressions from AST nodes.
7
8 This function converts AST nodes into a string representation of the expression they represent. It supports binary operations, unary operations, constants, identifiers, and function calls.
9
10 @param node: The AST node representing an expression.
11 @type node: pycparser.c_ast.Node
12 @param internalVariables: A dictionary to store and manage internal variables.
13 @type internalVariables: dict
14 @return: A string representing the reconstructed expression.
15 @rtype: str
16 @raises ValueError: If the node type is unsupported.
17 """
18 if isinstance(node, c_ast.BinaryOp):
19 leftExpr = reconstruct_expression(node.left,internalVariables)
20 rightExpr = reconstruct_expression(node.right,internalVariables)
21 return f"({leftExpr} {node.op} {rightExpr})"
22 elif isinstance(node, c_ast.ID):
23 return node.name
24 elif isinstance(node,c_ast.Constant):
25 return node.value
26 elif isinstance(node,c_ast.UnaryOp):
27 return "!" + reconstruct_expression(node.expr,internalVariables)
28 elif isinstance(node,c_ast.FuncCall):
29 if node.name.name == "reg":
30 varName = str(node.args.exprs[0].name) + "_reg";
31 if varName not in internalVariables:
32 internalVariables[varName] = {
33 "type" : "reg",
34 "value" : str(node.args.exprs[0].name)
35 }
36 return varName;
37 else:
38 raise ValueError(f"Unsupported node type: {type(node).__name__}")

42 """
43 @brief Handles declaration blocks in the AST and updates internal variables.
44
45 This function processes declaration nodes to determine the type of the variable (either "reg" or "wire") and reconstructs its initial value.
46
47 @param node: The declaration node being processed.
48 @type node: pycparser.c_ast.Decl
49 @param internalVariables: A dictionary to store and manage internal variables.
50 @type internalVariables: dict
51 @return: Boolean indicating if a register variable is present.
52 @rtype: bool
53 """
54 regPresent = False
55 if node.name not in internalVariables:
56 internalVariables[node.name] = {
57 "type" : "",
58 "value" : ""
59 }
60 if isinstance(node.init,c_ast.FuncCall) and node.init.name.name == "reg":
61 internalVariables[node.name]["type"] = "reg"
62 regPresent = True
63 exp = reconstruct_expression(node.init.args.exprs[0],internalVariables)
64 else:
65 internalVariables[node.name]["type"] = "wire"
66 exp = reconstruct_expression(node.init,internalVariables)
67 internalVariables[node.name]["value"] = exp
68 return regPresent

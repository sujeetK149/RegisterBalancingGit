43def extractOperands(node):
44 """
45 Extracts operands from a node, handling different node types.
46
47 @param node: The node from which to extract operands.
48
49 @return: A list of operands or a string representing the operand.
50 @raises ValueError: If the node type is unsupported.
51 """
52 if (isinstance(node, c_ast.BinaryOp)):
53 leftOperand = extractOperands(node.left)
54 rightOperand = extractOperands(node.right)
55 return [leftOperand,rightOperand]
56 elif isRegCall(node):
57 return "v_reg_" + node.args.exprs[0].name
58 elif isinstance(node,c_ast.ID):
59 return node.name
60 elif isinstance(node,c_ast.UnaryOp):
61 return "v_neg_" + extractOperands(node.expr)
62 else:
63 raise ValueError(f"Unsupported node type: {type(node).__name__}")
